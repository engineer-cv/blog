### 1、模板字符串的使用

   ```JavaScript
   //1、ES6之前拼接字符串和其他标识符
   const name = "why"
   const age = 18
   const height = 1.88
   console.log("my name is " + name + ", age is " + age + ", height is " + height)
   //2、ES6提供模板字符串 ``  
   //注:``这个单引号必须是tab键上面的那个键创建出来的
   const message = `my name is ${name}, age is ${age}, height is ${height}`    
   console.log(message)

   const info = `age double is ${age * 2}`
   console.log(info)

   function doubleAge() {
     return age * 2
   }

   const info2 = `double age is ${doubleAge()}`
   console.log(info2)
   ```

### 2、新增的调用函数的方式:标签模块字符串

   ```JavaScript
   // 第一个参数依然是模块字符串中整个字符串, 只是被切成多块,放到了一个数组中
   // 第二个参数是模块字符串中, 第一个 ${}, 以此类推
   // 如：foo`Hello${name}Wo${age}rld` ---> 相当于foo(['Hello','Wo','rld'],name,age)
   function foo(m, n, x) {
     console.log(m, n, x, '---------')
     console.log(m)
     console.log(n)
     console.log(x)
   }

   foo("Hello", "World")

   // 另外调用函数的方式: 标签模块字符串
   // foo``

   foo`Hello World`;//只传入了一个参数：['Hello World']
   const name = "why"
   const age = 18
   // // ['Hello', 'Wo', 'rld']
   foo`Hello${name}Wo${age}rld`//  相当于foo(['Hello','Wo','rld'],name,age)
   ```
